# Name router to vyos3
configure
set system host-name vyos3

# Delete old OSPF and interface IPs
delete protocols ospf
delete interfaces ethernet eth0 address 10.85.2.2/30
delete interfaces ethernet eth1 address 10.85.4.2/30
delete interfaces ethernet eth2 address 10.85.6.1/30

# Set new interface IPs based on the XY scheme
set interfaces ethernet eth0 address '10.85.13.2/30'
set interfaces ethernet eth1 address '10.85.23.2/30'
set interfaces ethernet eth2 address '10.85.34.1/30'

# Re-create OSPF with the new networks
set protocols ospf area 0 network '10.85.13.0/30'
set protocols ospf area 0 network '10.85.23.0/30'
set protocols ospf area 0 network '10.85.34.0/30'
set protocols ospf area 0 network '192.168.7.3/32'
set protocols ospf parameters router-id '192.168.7.3'

# Save configurations
commit
save
exit
